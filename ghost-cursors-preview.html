/* Ghost Cursor Pack — Recon_Set
   Drop these into your :root block, replacing the existing --cursor-* vars.
   Files go in: /public/cursors/ in your React project.
   ----------------------------------------------------------------------- */

:root {
  /* Normal ghost — default pointer */
  --cursor-default: url('/cursors/normal.cur') 4 4, auto;

  /* Glowing ghost with finger tip — links and buttons */
  --cursor-pointer: url('/cursors/link.cur') 4 4, pointer;

  /* Ghost + I-beam — text inputs */
  --cursor-text: url('/cursors/text.cur') 16 16, text;

  /* Ghost + ? badge — help elements */
  --cursor-help: url('/cursors/help.cur') 4 4, help;

  /* Ghost + four-way arrows — draggable elements */
  --cursor-move: url('/cursors/move.cur') 24 24, move;

  /* Faded ghost + circle-slash — disabled elements */
  --cursor-not-allowed: url('/cursors/unavailable.cur') 4 4, not-allowed;

  /* Ghost + small spinner — background loading */
  --cursor-working: url('/cursors/working.ani') 4 4, progress;

  /* Spinner ghost — blocking wait */
  --cursor-busy: url('/cursors/busy.ani') 16 16, wait;

  /* Crosshair ghost — precision / canvas */
  --cursor-crosshair: url('/cursors/cross.cur') 18 18, crosshair;
}

/* Apply to elements — matches your existing CSS selectors */
body { cursor: var(--cursor-default); }

a, button, [role="button"],
.tool-card, .nav-item, .bottom-nav-item,
.result-title, .hamburger-btn, .new-scan-btn,
.action-btn, .execute-btn, .settings-theme-card,
.settings-toggle, summary {
  cursor: var(--cursor-pointer);
}

input, textarea { cursor: var(--cursor-text); }
[disabled], :disabled { cursor: var(--cursor-not-allowed); opacity: 0.55; }
.scanning, [aria-busy="true"] { cursor: var(--cursor-busy); }

/* ── Theme-aware colour tinting via CSS filter ──────────────────────── */
/* Because .cur files are fixed colours, we use hue-rotate + saturate
   filters on elements that use cursors to shift the purple (#8B73D0)
   toward each theme accent. Applied on :root per data-theme.           */

[data-theme="phantom"]    { --cursor-hue: 150deg; --cursor-sat: 2; }
[data-theme="midnight"]   { --cursor-hue: 20deg;  --cursor-sat: 1.3; }
[data-theme="ocean"]      { --cursor-hue: 170deg; --cursor-sat: 1.8; }
[data-theme="synthwave"]  { --cursor-hue: 300deg; --cursor-sat: 1.5; }
[data-theme="crimson"]    { --cursor-hue: 340deg; --cursor-sat: 2; }
[data-theme="catppuccin"] { --cursor-hue: 10deg;  --cursor-sat: 0.9; }
[data-theme="light"]      { --cursor-hue: 200deg; --cursor-sat: 2.5; }

/* Note: CSS filter does not directly affect cursor images in most browsers.
   For full theme-coloured cursors, use the SVG inline cursor approach
   already in your frontend.css — the .cur files are the fallback/install pack. */
How to use in your project:
1. Place all .cur and .ani files inside public/cursors/ in your React project.
2. Replace the --cursor-* variables in your frontend.css with the ones above.
3. The SVG inline cursors (already in your CSS) handle theme colour changes live — the .cur files are the desktop-quality fallback when SVG cursors aren't supported.
4. Browser note: .ani animated cursors work in Chrome and Edge. Firefox falls back to the progress/wait system cursor.
